544FORCEINLINE VectorRegister VectorReciprocalSqrtAccurate(const VectorRegister& Vec)
545{
546 // Perform two passes of Newton-Raphson iteration on the hardware estimate
547 // v^-0.5 = x
548 // => x^2 = v^-1
549 // => 1/(x^2) = v
550 // => F(x) = x^-2 - v
551 // F'(x) = -2x^-3
552
553 // x1 = x0 - F(x0)/F'(x0)
554 // => x1 = x0 + 0.5 * (x0^-2 - Vec) * x0^3
555 // => x1 = x0 + 0.5 * (x0 - Vec * x0^3)
556 // => x1 = x0 + x0 * (0.5 - 0.5 * Vec * x0^2)
557
558 const VectorRegister OneHalf = GlobalVectorConstants::FloatOneHalf;
559 const VectorRegister VecDivBy2 = VectorMultiply(Vec, OneHalf);
560
561 // Initial estimate
562 const VectorRegister x0 = VectorReciprocalSqrt(Vec);
563
564 // First iteration
565 VectorRegister x1 = VectorMultiply(x0, x0);
566 x1 = VectorSubtract(OneHalf, VectorMultiply(VecDivBy2, x1));
567 x1 = VectorMultiplyAdd(x0, x1, x0);
568
569 // Second iteration
570 VectorRegister x2 = VectorMultiply(x1, x1);
571 x2 = VectorSubtract(OneHalf, VectorMultiply(VecDivBy2, x2));
572 x2 = VectorMultiplyAdd(x1, x2, x1);
573
574 return x2;
575}

583FORCEINLINE VectorRegister VectorReciprocalAccurate(const VectorRegister& Vec)
584{
585 // Perform two passes of Newton-Raphson iteration on the hardware estimate
586 // x1 = x0 - f(x0) / f'(x0)
587 //
588 // 1 / Vec = x
589 // => x * Vec = 1
590 // => F(x) = x * Vec - 1
591 // F'(x) = Vec
592 // => x1 = x0 - (x0 * Vec - 1) / Vec
593 //
594 // Since 1/Vec is what we're trying to solve, use an estimate for it, x0
595 // => x1 = x0 - (x0 * Vec - 1) * x0 = 2 * x0 - Vec * x0^2
596
597 // Initial estimate
598 const VectorRegister x0 = VectorReciprocal(Vec);
599
600 // First iteration
601 const VectorRegister x0Squared = VectorMultiply(x0, x0);
602 const VectorRegister x0Times2 = VectorAdd(x0, x0);
603 const VectorRegister x1 = VectorSubtract(x0Times2, VectorMultiply(Vec, x0Squared));
604
605 // Second iteration
606 const VectorRegister x1Squared = VectorMultiply(x1, x1);
607 const VectorRegister x1Times2 = VectorAdd(x1, x1);
608 const VectorRegister x2 = VectorSubtract(x1Times2, VectorMultiply(Vec, x1Squared));
609
610 return x2;
611}

654FORCEINLINE void VectorMatrixMultiply(void* Result, const void* Matrix1, const void* Matrix2)
655{
656 const VectorRegister* A = (const VectorRegister*)Matrix1;
657 const VectorRegister* B = (const VectorRegister*)Matrix2;
658 VectorRegister* R = (VectorRegister*)Result;
659 VectorRegister Temp, R0, R1, R2, R3;
660
661 // First row of result (Matrix1[0] * Matrix2).
662 Temp = VectorMultiply(VectorReplicate(A[0], 0), B[0]);
663 Temp = VectorMultiplyAdd(VectorReplicate(A[0], 1), B[1], Temp);
664 Temp = VectorMultiplyAdd(VectorReplicate(A[0], 2), B[2], Temp);
665 R0 = VectorMultiplyAdd(VectorReplicate(A[0], 3), B[3], Temp);
666
667 // Second row of result (Matrix1[1] * Matrix2).
668 Temp = VectorMultiply(VectorReplicate(A[1], 0), B[0]);
669 Temp = VectorMultiplyAdd(VectorReplicate(A[1], 1), B[1], Temp);
670 Temp = VectorMultiplyAdd(VectorReplicate(A[1], 2), B[2], Temp);
671 R1 = VectorMultiplyAdd(VectorReplicate(A[1], 3), B[3], Temp);
672
673 // Third row of result (Matrix1[2] * Matrix2).
674 Temp = VectorMultiply(VectorReplicate(A[2], 0), B[0]);
675 Temp = VectorMultiplyAdd(VectorReplicate(A[2], 1), B[1], Temp);
676 Temp = VectorMultiplyAdd(VectorReplicate(A[2], 2), B[2], Temp);
677 R2 = VectorMultiplyAdd(VectorReplicate(A[2], 3), B[3], Temp);
678
679 // Fourth row of result (Matrix1[3] * Matrix2).
680 Temp = VectorMultiply(VectorReplicate(A[3], 0), B[0]);
681 Temp = VectorMultiplyAdd(VectorReplicate(A[3], 1), B[1], Temp);
682 Temp = VectorMultiplyAdd(VectorReplicate(A[3], 2), B[2], Temp);
683 R3 = VectorMultiplyAdd(VectorReplicate(A[3], 3), B[3], Temp);
684
685 // Store result
686 R[0] = R0;
687 R[1] = R1;
688 R[2] = R2;
689 R[3] = R3;
690}

710FORCEINLINE VectorRegister VectorTransformVector(const VectorRegister& VecP, const void* MatrixM)
711{
712 const VectorRegister* M = (const VectorRegister*)MatrixM;
713 VectorRegister VTempX, VTempY, VTempZ, VTempW;
714
715 // Splat x,y,z and w
716 VTempX = VectorReplicate(VecP, 0);
717 VTempY = VectorReplicate(VecP, 1);
718 VTempZ = VectorReplicate(VecP, 2);
719 VTempW = VectorReplicate(VecP, 3);
720 // Mul by the matrix
721 VTempX = VectorMultiply(VTempX, M[0]);
722 VTempY = VectorMultiply(VTempY, M[1]);
723 VTempZ = VectorMultiply(VTempZ, M[2]);
724 VTempW = VectorMultiply(VTempW, M[3]);
725 // Add them all together
726 VTempX = VectorAdd(VTempX, VTempY);
727 VTempZ = VectorAdd(VTempZ, VTempW);
728 VTempX = VectorAdd(VTempX, VTempZ);
729
730 return VTempX;
731}

1075FORCEINLINE bool VectorContainsNaNOrInfinite(const VectorRegister& Vec)
1076{
1077 // https://en.wikipedia.org/wiki/IEEE_754-1985
1078 // Infinity is represented with all exponent bits set, with the correct sign bit.
1079 // NaN is represented with all exponent bits set, plus at least one fraction/significand bit set.
1080 // This means finite values will not have all exponent bits set, so check against those bits.
1081
1082 union { float F; uint32 U; } InfUnion;
1083 InfUnion.U = 0x7F800000;
1084 const float Inf = InfUnion.F;
1085 const VectorRegister FloatInfinity = MakeVectorRegister(Inf, Inf, Inf, Inf);
1086
1087 // Mask off Exponent
1088 VectorRegister ExpTest = VectorBitwiseAnd(Vec, FloatInfinity);
1089 // Compare to full exponent. If any are full exponent (not finite), the signs copied to the mask are non-zero, otherwise it's zero and finite.
1090 bool IsFinite = VectorMaskBits(VectorCompareEQ(ExpTest, FloatInfinity)) == 0;
1091 return !IsFinite;
1092}

1189 //based on a good discussion here http://forum.devmaster.net/t/fast-and-accurate-sine-cosine/9648

1191 //Average error of 0.000128
1192 //Max error of 0.001091
1193

1195 y = VectorSubtract(y, VectorFloor(VectorAdd(y, GlobalVectorConstants::FloatOneHalf)));
1196 y = VectorMultiply(VectorSinConstantsSSE::A, VectorMultiply(y, VectorSubtract(GlobalVectorConstants::FloatOneHalf, VectorAbs(y))));

1213FORCEINLINE void VectorSinCos(VectorRegister* RESTRICT VSinAngles, VectorRegister* RESTRICT VCosAngles, const VectorRegister* RESTRICT VAngles)
1214{
1215 // Map to [-pi, pi]
1216 // X = A - 2pi * round(A/2pi)
1217 // Note the round(), not truncate(). In this case round() can round halfway cases using round-to-nearest-even OR round-to-nearest.
1218
1219 // Quotient = round(A/2pi)
1220 VectorRegister Quotient = VectorMultiply(*VAngles, GlobalVectorConstants::OneOverTwoPi);
1221 Quotient = _mm_cvtepi32_ps(_mm_cvtps_epi32(Quotient)); // round to nearest even is the default rounding mode but that's fine here.
1222 // X = A - 2pi * Quotient
1223 VectorRegister X = VectorSubtract(*VAngles, VectorMultiply(GlobalVectorConstants::TwoPi, Quotient));
1224
1225 // Map in [-pi/2,pi/2]
1226 VectorRegister sign = VectorBitwiseAnd(X, GlobalVectorConstants::SignBit);
1227 VectorRegister c = VectorBitwiseOr(GlobalVectorConstants::Pi, sign); // pi when x >= 0, -pi when x < 0
1228 VectorRegister absx = VectorAbs(X);
1229 VectorRegister rflx = VectorSubtract(c, X);
1230 VectorRegister comp = VectorCompareGT(absx, GlobalVectorConstants::PiByTwo);
1231 X = VectorSelect(comp, rflx, X);
1232 sign = VectorSelect(comp, GlobalVectorConstants::FloatMinusOne, GlobalVectorConstants::FloatOne);
1233
1234 const VectorRegister XSquared = VectorMultiply(X, X);
1235
1236 // 11-degree minimax approximation
1237 //*ScalarSin = (((((-2.3889859e-08f * y2 + 2.7525562e-06f) * y2 - 0.00019840874f) * y2 + 0.0083333310f) * y2 - 0.16666667f) * y2 + 1.0f) * y;
1238 const VectorRegister SinCoeff0 = MakeVectorRegister(1.0f, -0.16666667f, 0.0083333310f, -0.00019840874f);
1239 const VectorRegister SinCoeff1 = MakeVectorRegister(2.7525562e-06f, -2.3889859e-08f, /*unused*/ 0.f, /*unused*/ 0.f);
1240
1241 VectorRegister S;
1242 S = VectorReplicate(SinCoeff1, 1);
1243 S = VectorMultiplyAdd(XSquared, S, VectorReplicate(SinCoeff1, 0));
1244 S = VectorMultiplyAdd(XSquared, S, VectorReplicate(SinCoeff0, 3));
1245 S = VectorMultiplyAdd(XSquared, S, VectorReplicate(SinCoeff0, 2));
1246 S = VectorMultiplyAdd(XSquared, S, VectorReplicate(SinCoeff0, 1));
1247 S = VectorMultiplyAdd(XSquared, S, VectorReplicate(SinCoeff0, 0));
1248 *VSinAngles = VectorMultiply(S, X);
1249
1250 // 10-degree minimax approximation
1251 //*ScalarCos = sign * (((((-2.6051615e-07f * y2 + 2.4760495e-05f) * y2 - 0.0013888378f) * y2 + 0.041666638f) * y2 - 0.5f) * y2 + 1.0f);
1252 const VectorRegister CosCoeff0 = MakeVectorRegister(1.0f, -0.5f, 0.041666638f, -0.0013888378f);
1253 const VectorRegister CosCoeff1 = MakeVectorRegister(2.4760495e-05f, -2.6051615e-07f, /*unused*/ 0.f, /*unused*/ 0.f);
1254
1255 VectorRegister C;
1256 C = VectorReplicate(CosCoeff1, 1);
1257 C = VectorMultiplyAdd(XSquared, C, VectorReplicate(CosCoeff1, 0));
1258 C = VectorMultiplyAdd(XSquared, C, VectorReplicate(CosCoeff0, 3));
1259 C = VectorMultiplyAdd(XSquared, C, VectorReplicate(CosCoeff0, 2));
1260 C = VectorMultiplyAdd(XSquared, C, VectorReplicate(CosCoeff0, 1));
1261 C = VectorMultiplyAdd(XSquared, C, VectorReplicate(CosCoeff0, 0));
1262 *VCosAngles = VectorMultiply(C, sign);
1263}
