18/*
19 Is type DerivedType inherited from BaseType.
20*/
21template<typename DerivedType, typename BaseType>
22struct TIsDerivedFrom
23{
24 // Different size types so we can compare their sizes later.
25 typedef char No[1];
26 typedef char Yes[2];
27
28 // Overloading Test() s.t. only calling it with something that is
29 // a BaseType (or inherited from the BaseType) will return a Yes.
30 static Yes& Test(BaseType*);
31 static Yes& Test(const BaseType*);
32 static No& Test(...);
33
34 // Makes a DerivedType ptr.
35 static DerivedType* DerivedTypePtr() { return nullptr; }
36
37public:
38
39 // Test the derived type pointer. If it inherits from BaseType, the Test( BaseType* )
40 // will be chosen. If it does not, Test( ... ) will be chosen.
41 static const bool Value = sizeof(Test(DerivedTypePtr())) == sizeof(Yes);
42
43 static const bool IsDerived = Value;
44};

120/*
121 TFormatSpecifier, only applies to numeric types.
122*/
123template<typename T>
124struct TFormatSpecifier
125{
126 FORCEINLINE static TCHAR const* GetFormatSpecifier()
127 {
128 // Force the template instantiation to be dependent upon T so the compiler cannot automatically decide that this template can never be instantiated.
129 // If the static_assert below were a constant 0 or something not dependent on T, compilers are free to detect this and fail to compile the template.
130 // As specified in the C++ standard s14.6p8. A compiler is free to give a diagnostic here or not. MSVC ignores it, and clang/gcc instantiates the
131 // template and triggers the static_assert.
132 static_assert(sizeof(T) < 0, "Format specifier not supported for this type."); // request for a format specifier for a type we do not know about
133 return TEXT("Unknown");
134 }
135};
136#define Expose_TFormatSpecifier(type, format) \
137template<> \
138struct TFormatSpecifier<type> \
139{ \
140 FORCEINLINE static const TCHAR (&GetFormatSpecifier())[5] \
141 { \
142 static const TCHAR Spec[5] = TEXT(format); \
143 return Spec; \
144 } \
145};

229/*
230 Tests if a type T is bitwise-constructible from a given argument type U.
231 That is, whether or not the U can be memcpy'd in order to produce an instance of T, rather than having to go via a constructor.
232
233 Examples:
234 TIsBitwiseConstructible<PODType, PODType >::Value == true // PODs can be trivially copied
235 TIsBitwiseConstructible<const int*, int* >::Value == true // a non-const Derived pointer is trivially copyable as a const Base pointer
236 TIsBitwiseConstructible<int*, const int*>::Value == false // not legal the other way because it would be a const-correctness violation
237 TIsBitwiseConstructible<int32, uint32 >::Value == true // signed integers can be memcpy'd as unsigned integers
238 TIsBitwiseConstructible<uint32, int32 >::Value == true // and vice versa
239*/
240
241template<typename T, typename Arg>
242struct TIsBitwiseConstructible
243{
244 static_assert(
245 !TIsReferenceType<T >::Value &&
246 !TIsReferenceType<Arg>::Value,
247 "TIsBitwiseConstructible is not designed to accept reference types");
248
249 static_assert(
250 TAreTypesEqual<T, typename TRemoveCV<T >::Type>::Value &&
251 TAreTypesEqual<Arg, typename TRemoveCV<Arg>::Type>::Value,
252 "TIsBitwiseConstructible is not designed to accept qualified types");
253
254 // Assume no bitwise construction in general
255 enum { Value = false };
256};
257
258template<typename T>
259struct TIsBitwiseConstructible<T, T>
260{
261 // Ts can always be bitwise constructed from itself if it is trivially copyable.
262 enum { Value = TIsTriviallyCopyConstructible<T>::Value };
263};
264
265template<typename T, typename U>
266struct TIsBitwiseConstructible<const T, U> : TIsBitwiseConstructible<T, U>
267{
268 // Constructing a const T is the same as constructing a T
269};
270
271/*
272 Const pointers can be bitwise constructed from non-const pointers.
273 This is not true for pointer conversions in general, e.g. where an offset may need to be applied in the case
274 of multiple inheritance, but there is no way of detecting that at compile-time.
275*/
276template<typename T>
277struct TIsBitwiseConstructible<const T*, T*>
278{
279 // Constructing a const T is the same as constructing a T
280 enum { Value = true };
281};
282

284// (assuming two's-complement, which we are)
