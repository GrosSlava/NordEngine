14// Returns a normalized 4 vector = Vector / |Vector|.
15// There is no handling of zero length vectors, use VectorNormalizeSafe if this is a possible input.
16FORCEINLINE VectorRegister VectorNormalizeAccurate(const VectorRegister& Vector)
17{
18 const VectorRegister SquareSum = VectorDot4(Vector, Vector);
19 const VectorRegister InvLength = VectorReciprocalSqrtAccurate(SquareSum);
20 const VectorRegister NormalizedVector = VectorMultiply(InvLength, Vector);
21 return NormalizedVector;
22}

24// Returns ((Vector dot Vector) >= 1e-8) ? (Vector / |Vector|) : DefaultValue
25// Uses accurate 1/sqrt, not the estimate
26FORCEINLINE VectorRegister VectorNormalizeSafe(const VectorRegister& Vector, const VectorRegister& DefaultValue)
27{
28 const VectorRegister SquareSum = VectorDot4(Vector, Vector);
29 const VectorRegister NonZeroMask = VectorCompareGE(SquareSum, GlobalVectorConstants::SmallLengthThreshold);
30 const VectorRegister InvLength = VectorReciprocalSqrtAccurate(SquareSum);
31 const VectorRegister NormalizedVector = VectorMultiply(InvLength, Vector);
32 return VectorSelect(NonZeroMask, NormalizedVector, DefaultValue);
33}

72// A and B are quaternions. The result is A + (|A.B| >= 0 ? 1 : -1) * B
73FORCEINLINE VectorRegister VectorAccumulateQuaternionShortestPath(const VectorRegister& A, const VectorRegister& B)
74{
75 // Blend rotation
76 // To ensure the 'shortest route', we make sure the dot product between the both rotations is positive.
77 // const float Bias = (|A.B| >= 0 ? 1 : -1)
78 // return A + B * Bias;
79 const VectorRegister Zero = VectorZero();
80 const VectorRegister RotationDot = VectorDot4(A, B);
81 const VectorRegister QuatRotationDirMask = VectorCompareGE(RotationDot, Zero);
82 const VectorRegister NegativeB = VectorSubtract(Zero, B);
83 const VectorRegister BiasTimesB = VectorSelect(QuatRotationDirMask, B, NegativeB);
84 return VectorAdd(A, BiasTimesB);
85}

93// Normalize Rotator
94FORCEINLINE VectorRegister VectorNormalizeRotator(const VectorRegister& UnnormalizedRotator)
95{
96 // shift in the range [-360,360]
97 VectorRegister V0 = VectorMod(UnnormalizedRotator, GlobalVectorConstants::Float360);
98 VectorRegister V1 = VectorAdd(V0, GlobalVectorConstants::Float360);
99 VectorRegister V2 = VectorSelect(VectorCompareGE(V0, VectorZero()), V0, V1);
100
101 // shift to [-180,180]
102 VectorRegister V3 = VectorSubtract(V2, GlobalVectorConstants::Float360);
103 VectorRegister V4 = VectorSelect(VectorCompareGT(V2, GlobalVectorConstants::Float180), V3, V2);
104
105 return V4;
106}

112FORCEINLINE VectorRegister VectorLerpQuat(const VectorRegister& A, const VectorRegister& B, const VectorRegister& Alpha)
113{
114 // Blend rotation
115 // To ensure the 'shortest route', we make sure the dot product between the both rotations is positive.
116 // const float Bias = (|A.B| >= 0 ? 1 : -1)
117 // Rotation = (B * Alpha) + (A * (Bias * (1.f - Alpha)));
118 const VectorRegister Zero = VectorZero();
119
120 const VectorRegister OneMinusAlpha = VectorSubtract(VectorOne(), Alpha);
121
122 const VectorRegister RotationDot = VectorDot4(A, B);
123 const VectorRegister QuatRotationDirMask = VectorCompareGE(RotationDot, Zero);
124 const VectorRegister NegativeA = VectorSubtract(Zero, A);
125 const VectorRegister BiasTimesA = VectorSelect(QuatRotationDirMask, A, NegativeA);
126 const VectorRegister BTimesWeight = VectorMultiply(B, Alpha);
127 const VectorRegister UnnormalizedResult = VectorMultiplyAdd(BiasTimesA, OneMinusAlpha, BTimesWeight);
128
129 return UnnormalizedResult;
130}

159FORCEINLINE VectorRegister VectorQuaternionRotateVector(const VectorRegister& Quat, const VectorRegister& VectorW0)
160{
161 // Q * V * Q.Inverse
162 //const VectorRegister InverseRotation = VectorQuaternionInverse(Quat);
163 //const VectorRegister Temp = VectorQuaternionMultiply2(Quat, VectorW0);
164 //const VectorRegister Rotated = VectorQuaternionMultiply2(Temp, InverseRotation);
165
166 // Equivalence of above can be shown to be:
167 // http://people.csail.mit.edu/bkph/articles/Quaternions.pdf
168 // V' = V + 2w(Q x V) + (2Q x (Q x V))
169 // refactor:
170 // V' = V + w(2(Q x V)) + (Q x (2(Q x V)))
171 // T = 2(Q x V);
172 // V' = V + w*(T) + (Q x T)
173
174 const VectorRegister QW = VectorReplicate(Quat, 3);
175 VectorRegister T = VectorCross(Quat, VectorW0);
176 T = VectorAdd(T, T);
177 const VectorRegister VTemp0 = VectorMultiplyAdd(QW, T, VectorW0);
178 const VectorRegister VTemp1 = VectorCross(Quat, T);
179 const VectorRegister Rotated = VectorAdd(VTemp0, VTemp1);
180 return Rotated;
181}
