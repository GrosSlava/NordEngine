265 FORCEINLINE float GetMaximumAxisScale() const noexcept
266 {
267 float Scale3DAbsMax;
268 // Scale3DAbsXYZ1 = { Abs(X), Abs(Y)), Abs(Z), 0 }
269 const VectorRegister Scale3DAbsXYZ0 = VectorAbs(Scale3D);
270 // Scale3DAbsYZX1 = { Abs(Y),Abs(Z)),Abs(X), 0 }
271 const VectorRegister Scale3DAbsYZX0 = VectorSwizzle(Scale3DAbsXYZ0, 1, 2, 0, 3);
272 // Scale3DAbsZXY1 = { Abs(Z),Abs(X)),Abs(Y), 0 }
273 const VectorRegister Scale3DAbsZXY0 = VectorSwizzle(Scale3DAbsXYZ0, 2, 0, 1, 3);
274 // t0 = { Max(Abs(X), Abs(Y)), Max(Abs(Y), Abs(Z)), Max(Abs(Z), Abs(X)), 0 }
275 const VectorRegister t0 = VectorMax(Scale3DAbsXYZ0, Scale3DAbsYZX0);
276 // t1 = { Max(Abs(X), Abs(Y), Abs(Z)), Max(Abs(Y), Abs(Z), Abs(X)), Max(Abs(Z), Abs(X), Abs(Y)), 0 }
277 const VectorRegister t2 = VectorMax(t0, Scale3DAbsZXY0);
278 // Scale3DAbsMax = Max(Abs(X), Abs(Y), Abs(Z));
279 VectorStoreFloat1(t2, &Scale3DAbsMax);
280
281 return Scale3DAbsMax;
282 }

309 FORCEINLINE FVector4D TransformFVector4(const FVector4D& V) const noexcept
310 {
311 const VectorRegister InputVector = VectorLoadAligned(&V);
312
313 //Transform using QST is following
314 //QST(P) = Q.Rotate(S*P) + T where Q = quaternion, S = scale, T = translation
315
316 //RotatedVec = Q.Rotate(Scale*V.X, Scale*V.Y, Scale*V.Z, 0.f)
317 const VectorRegister InputVectorW0 = VectorSet_W0(InputVector);
318 const VectorRegister ScaledVec = VectorMultiply(Scale3D, InputVectorW0);
319 const VectorRegister RotatedVec = VectorQuaternionRotateVector(Rotation, ScaledVec);
320
321 // NewVect.XYZ += Translation * W
322 // NewVect.W += 1 * W
323 const VectorRegister WWWW = VectorReplicate(InputVector, 3);
324 const VectorRegister TranslatedVec = VectorMultiplyAdd(Translation, WWWW, RotatedVec);
325
326 FVector4D NewVectOutput;
327 VectorStoreAligned(TranslatedVec, &NewVectOutput);
328 return NewVectOutput;
329
330 }

391 FORCEINLINE FVector3D InverseTransformPosition(const FVector3D& V) const noexcept
392 {
393 const VectorRegister InputVector = VectorLoadFloat3_W0(&V);
394
395 // (V-Translation)
396 const VectorRegister TranslatedVec = VectorSet_W0(VectorSubtract(InputVector, Translation));
397
398 // ( Rotation.Inverse() * (V-Translation) )
399 const VectorRegister VR = VectorQuaternionInverseRotateVector(Rotation, TranslatedVec);
400
401 // GetSafeScaleReciprocal(Scale3D);
402 const VectorRegister SafeReciprocal = GetSafeScaleReciprocal(Scale3D);
403
404 // ( Rotation.Inverse() * (V-Translation) ) * GetSafeScaleReciprocal(Scale3D);
405 const VectorRegister VResult = VectorMultiply(VR, SafeReciprocal);
406
407 FVector3D Result;
408 VectorStoreFloat3(VResult, &Result);
409 return Result;

770 FORCEINLINE void Accumulate(const FTransform& SourceAtom) noexcept
771 {
772 const VectorRegister BlendedRotation = SourceAtom.Rotation;
773 const VectorRegister RotationW = VectorReplicate(BlendedRotation, 3);
774
775 // if( Square(SourceAtom.Rotation.W) < 1.f - DELTA * DELTA )
776 if (VectorAnyGreaterThan(GlobalVectorConstants::RotationSignificantThreshold, VectorMultiply(RotationW, RotationW)))
777 {
778 // Rotation = SourceAtom.Rotation * Rotation;
779 Rotation = VectorQuaternionMultiply2(BlendedRotation, Rotation);
780 }
781
782 // Translation += SourceAtom.Translation;
783 // Scale *= SourceAtom.Scale;
784 Translation = VectorAdd(Translation, SourceAtom.Translation);
785 Scale3D = VectorMultiply(Scale3D, SourceAtom.Scale3D);
786 }

801 FORCEINLINE void Accumulate(const FTransform& Atom, const ScalarRegister& BlendWeight) noexcept
802 {
803 // SourceAtom = Atom * BlendWeight;
804 const VectorRegister BlendedRotation = VectorMultiply(Atom.Rotation, BlendWeight.Value);
805 const VectorRegister BlendedTranslation = VectorMultiply(Atom.Translation, BlendWeight.Value);
806 const VectorRegister BlendedScale = VectorMultiply(Atom.Scale3D, BlendWeight.Value);
807
808 const VectorRegister RotationW = VectorReplicate(BlendedRotation, 3);
809
810 // Add ref pose relative animation to base animation, only if rotation is significant.
811 // if( Square(SourceAtom.Rotation.W) < 1.f - DELTA * DELTA )
812 if (VectorAnyGreaterThan(GlobalVectorConstants::RotationSignificantThreshold, VectorMultiply(RotationW, RotationW)))
813 {
814 // Rotation = SourceAtom.Rotation * Rotation;
815 Rotation = VectorQuaternionMultiply2(BlendedRotation, Rotation);
816 }
817
818 // Translation += SourceAtom.Translation;
819 // Scale *= SourceAtom.Scale;
820 Translation = VectorAdd(Translation, BlendedTranslation);
821 Scale3D = VectorMultiply(Scale3D, BlendedScale);
822 }
833 FORCEINLINE void AccumulateWithShortestRotation(const FTransform& DeltaAtom, const ScalarRegister& BlendWeight) noexcept
834 {
835 const VectorRegister BlendedRotation = VectorMultiply(DeltaAtom.Rotation, BlendWeight.Value);
836

1132 FORCEINLINE void ToMatrixInternal(VectorRegister& OutDiagonals, VectorRegister& OutAdds, VectorRegister& OutSubtracts) const
1133 {
1134 const VectorRegister RotationX2Y2Z2 = VectorAdd(Rotation, Rotation); // x2, y2, z2
1135 const VectorRegister RotationXX2YY2ZZ2 = VectorMultiply(RotationX2Y2Z2, Rotation); // xx2, yy2, zz2
1136
1137 // The diagonal terms of the rotation matrix are:
1138 // (1 - (yy2 + zz2)) * scale
1139 // (1 - (xx2 + zz2)) * scale
1140 // (1 - (xx2 + yy2)) * scale
1141 const VectorRegister yy2_xx2_xx2 = VectorSwizzle(RotationXX2YY2ZZ2, 1, 0, 0, 0);
1142 const VectorRegister zz2_zz2_yy2 = VectorSwizzle(RotationXX2YY2ZZ2, 2, 2, 1, 0);
1143 const VectorRegister DiagonalSum = VectorAdd(yy2_xx2_xx2, zz2_zz2_yy2);
1144 const VectorRegister Diagonals = VectorSubtract(VectorOne(), DiagonalSum);
1145 OutDiagonals = VectorMultiply(Diagonals, Scale3D);
1146
1147 // Grouping the non-diagonal elements in the rotation block by operations:
1148 // ((x*y2,y*z2,x*z2) + (w*z2,w*x2,w*y2)) * scale.xyz and
1149 // ((x*y2,y*z2,x*z2) - (w*z2,w*x2,w*y2)) * scale.yxz
1150 // Rearranging so the LHS and RHS are in the same order as for +
1151 // ((x*y2,y*z2,x*z2) - (w*z2,w*x2,w*y2)) * scale.yxz
1152
1153 // RotBase = x*y2, y*z2, x*z2
1154 // RotOffset = w*z2, w*x2, w*y2
1155 const VectorRegister x_y_x = VectorSwizzle(Rotation, 0, 1, 0, 0);
1156 const VectorRegister y2_z2_z2 = VectorSwizzle(RotationX2Y2Z2, 1, 2, 2, 0);
1157 const VectorRegister RotBase = VectorMultiply(x_y_x, y2_z2_z2);
1158
1159 const VectorRegister w_w_w = VectorReplicate(Rotation, 3);
1160 const VectorRegister z2_x2_y2 = VectorSwizzle(RotationX2Y2Z2, 2, 0, 1, 0);
1161 const VectorRegister RotOffset = VectorMultiply(w_w_w, z2_x2_y2);
1162
1163 // Adds = (RotBase + RotOffset)*Scale3D : (x*y2 + w*z2) * Scale3D.X , (y*z2 + w*x2) * Scale3D.Y, (x*z2 + w*y2) * Scale3D.Z
1164 // Subtracts = (RotBase - RotOffset)*Scale3DYZX : (x*y2 - w*z2) * Scale3D.Y , (y*z2 - w*x2) * Scale3D.Z, (x*z2 - w*y2) * Scale3D.X
1165 const VectorRegister Adds = VectorAdd(RotBase, RotOffset);
1166 OutAdds = VectorMultiply(Adds, Scale3D);
1167 const VectorRegister Scale3DYZXW = VectorSwizzle(Scale3D, 1, 2, 0, 3);
1168 const VectorRegister Subtracts = VectorSubtract(RotBase, RotOffset);
1169 OutSubtracts = VectorMultiply(Subtracts, Scale3DYZXW);
1170 }

1209 FORCEINLINE FTransform InverseFast() const
1210 {
1211 // Inverse QST (A) = QST (~A)
1212 // Since A*~A = Identity,
1213 // A(P) = Q(A)*S(A)*P*-Q(A) + T(A)
1214 // ~A(A(P)) = Q(~A)*S(~A)*(Q(A)*S(A)*P*-Q(A) + T(A))*-Q(~A) + T(~A) = Identity
1215 // Q(~A)*Q(A)*S(~A)*S(A)*P*-Q(A)*-Q(~A) + Q(~A)*S(~A)*T(A)*-Q(~A) + T(~A) = Identity
1216 // [Q(~A)*Q(A)]*[S(~A)*S(A)]*P*-[Q(~A)*Q(A)] + [Q(~A)*S(~A)*T(A)*-Q(~A) + T(~A)] = I
1217
1218 // Identity Q = (0, 0, 0, 1) = Q(~A)*Q(A)
1219 // Identity Scale = 1 = S(~A)*S(A)
1220 // Identity Translation = (0, 0, 0) = [Q(~A)*S(~A)*T(A)*-Q(~A) + T(~A)]
1221
1222 // Q(~A) = Q(~A)
1223 // S(~A) = 1.f/S(A)
1224 // T(~A) = - (Q(~A)*S(~A)*T(A)*Q(A))
1225
1226 if (!IsRotationNormalized() || !VectorAnyGreaterThan(VectorAbs(Scale3D), GlobalVectorConstants::SmallNumber)) return *this;
1227
1228 // Invert the scale
1229 const VectorRegister InvScale = VectorSet_W0(GetSafeScaleReciprocal(VectorSet_W1(Scale3D), ScalarRegister(GlobalVectorConstants::SmallNumber)));
1230
1231 // Invert the rotation
1232 const VectorRegister InvRotation = VectorQuaternionInverse(Rotation);
1233
1234 // Invert the translation
1235 const VectorRegister ScaledTranslation = VectorMultiply(InvScale, Translation);
1236 const VectorRegister t2 = VectorQuaternionRotateVector(InvRotation, ScaledTranslation);
1237 const VectorRegister InvTranslation = VectorSet_W0(VectorNegate(t2));
1238
1239 return FTransform(InvRotation, InvTranslation, InvScale);
1240 }
